274 unsigned int Algorithm::intersection_number(sibling_iterator from1, sibling_iterator to1,
275  sibling_iterator from2, sibling_iterator to2,
276  BinaryPredicate fun) const
277  {
278  unsigned int ret=0;
279  sibling_iterator it1=from1;
280  while(it1!=to1) {
281  sibling_iterator it2=from2;
282  while(it2!=to2) {
283  if(fun(*it1,*it2))
284  ++ret;
285  ++it2;
286  }
287  ++it1;
288  }
289  return ret;
290  }
