253 unsigned int Algorithm::intersection_number(sibling_iterator from1, sibling_iterator to1,
254  sibling_iterator from2, sibling_iterator to2,
255  BinaryPredicate fun) const
256  {
257  unsigned int ret=0;
258  sibling_iterator it1=from1;
259  while(it1!=to1) {
260  sibling_iterator it2=from2;
261  while(it2!=to2) {
262  if(fun(*it1,*it2))
263  ++ret;
264  ++it2;
265  }
266  ++it1;
267  }
268  return ret;
269  }
