258 unsigned int Algorithm::intersection_number(sibling_iterator from1, sibling_iterator to1,
259  sibling_iterator from2, sibling_iterator to2,
260  BinaryPredicate fun) const
261  {
262  unsigned int ret=0;
263  sibling_iterator it1=from1;
264  while(it1!=to1) {
265  sibling_iterator it2=from2;
266  while(it2!=to2) {
267  if(fun(*it1,*it2))
268  ++ret;
269  ++it2;
270  }
271  ++it1;
272  }
273  return ret;
274  }
