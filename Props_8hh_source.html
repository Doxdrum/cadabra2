297  serialnum=0;
298  while(pm.first!=pm.second) {
299  if((*pm.first).second==(*walk).second.first)

303  }
304  }

306  }

312  }

314  }

411 const T* Properties::get_composite(Ex::iterator it1, Ex::iterator it2, int& serialnum1, int& serialnum2, bool ignore_parent_rel) const
412  {
413  const T* ret1=0;
414  const T* ret2=0;
415  bool found=false;
416 
417  bool inherits1=false, inherits2=false;
418  std::pair<property_map_t::const_iterator, property_map_t::const_iterator> pit1=props.equal_range(it1->name);
419  std::pair<property_map_t::const_iterator, property_map_t::const_iterator> pit2=props.equal_range(it2->name);
420 
421  property_map_t::const_iterator walk1=pit1.first;
422  while(walk1!=pit1.second) {
423  if((*walk1).second.first->match(*this, it1, ignore_parent_rel)) { // match for object 1 found
424  ret1=dynamic_cast<const T *>((*walk1).second.second);
425  if(ret1) { // property of the right type found for object 1
426  property_map_t::const_iterator walk2=pit2.first;
427  while(walk2!=pit2.second) {
428  if((*walk2).second.first->match(*this, it2, ignore_parent_rel)) { // match for object 1 found
429  ret2=dynamic_cast<const T *>((*walk2).second.second);
430  if(ret2) { // property of the right type found for object 2
431  if(ret1==ret2 && walk1!=walk2) { // accept if properties are the same and patterns are not
432  serialnum1=serial_number( (*walk1).second.second, (*walk1).second.first );
433  serialnum2=serial_number( (*walk2).second.second, (*walk2).second.first );
434  found=true;
435  goto done;
436  }
437  }
438  }
439  if(dynamic_cast<const PropertyInherit *>((*walk2).second.second))
440  inherits2=true;
441  ++walk2;
442  }
443  }
444  if(dynamic_cast<const PropertyInherit *>((*walk1).second.second))
445  inherits1=true;
446  }
447  ++walk1;
448  }
449 
450  // If no property was found, figure out whether a property is inherited from a child node.
451  if(!found && (inherits1 || inherits2)) {
452  Ex::sibling_iterator sib1, sib2;
453  if(inherits1) sib1=it1.begin();
454  else sib1=it1;
455  bool keepgoing1=true;
456  do { // 1
457  bool keepgoing2=true;
458  if(inherits2) sib2=it2.begin();
459  else sib2=it2;
460  do { // 2
461  const T* tmp=get_composite<T>((Ex::iterator)(sib1), (Ex::iterator)(sib2), serialnum1, serialnum2, ignore_parent_rel);
462  if(tmp) {
463  ret1=tmp;
464  found=true;
465  goto done;
466  }
467  if(!inherits2 || ++sib2==it2.end())
468  keepgoing2=false;
469  } while(keepgoing2);
470  if(!inherits1 || ++sib1==it1.end())
471  keepgoing1=false;
472  } while(keepgoing1);
473  }
474 
475  done:
476  if(!found) ret1=0;
477  return ret1;
478  }
